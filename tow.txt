function useGPS(){
  if(!navigator.geolocation){showToast('GPS NOT SUPPORTED','e');return;}
  document.getElementById('gpsOption').style.display='none';
  document.getElementById('gpsLoading').classList.add('show');
  navigator.geolocation.getCurrentPosition(
    pos=>{
      document.getElementById('gpsLoading').classList.remove('show');
      document.getElementById('gpsOption').style.display='flex';
      const lat=pos.coords.latitude.toFixed(4),lng=pos.coords.longitude.toFixed(4);
      document.getElementById('locValue').textContent='Current Location';
      closeLocDropdown();
      showToast('📡 GPS ACQUIRED · '+lat+'°N, '+lng+'°E','s');
    },
    err=>{
      document.getElementById('gpsLoading').classList.remove('show');
      document.getElementById('gpsOption').style.display='flex';
      showToast('GPS DENIED — USE MANUAL','e');
      toggleManualInput();
    },
    {timeout:8000}
  );
}

function getFiltered(){return PLACES.filter(p=>{
  if(evMode&&!p.ev)return false;
  if(activeFilter==='available'&&p.free===0)return false;
  if(!['all','available'].includes(activeFilter)&&!p.type.includes(activeFilter))return false;
  if(searchQ&&!p.name.toLowerCase().includes(searchQ)&&!p.location.toLowerCase().includes(searchQ))return false;
  return true;
});}
function getSorted(a){const c=[...a];if(sortMode===1)c.sort((a,b)=>parseInt(a.price)-parseInt(b.price));else if(sortMode===2)c.sort((a,b)=>b.rating-a.rating);else if(sortMode===3)c.sort((a,b)=>(b.free/b.total)-(a.free/a.total));return c;}
